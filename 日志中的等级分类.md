    static char *GetLogLevel(int iLevel)
    {
        if (iLevel == -1) {
            return "ERROR";
        } else if (iLevel == 0) {
            return "WARNING";
        } else if (iLevel == 1) {
            return "INFO";
        } else {
            return "DEBUG";
        }
    }
